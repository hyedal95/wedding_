# JS에서 새로운 방식
## 클래스 객체를 생성하여 사요한다.
    - = 안 사용
    - , 안 사용
    

```js
(function($){
    class Obj {
        init(){

            this.header();
            this.section1();
            this.section2();
            this.section3();
            this.footer();
        }
        header(){
            console.log('header');
        }
        section1(){
            console.log('section1');
        }
        section2(){
            console.log('section2');
        }
        section3(){

        }
        footer(){
            console.log('header');
        }
    }
    const obj = new Obj(); //클래스 객체를 생성하여 사용한다.
    obj.init();

})(jQuery);

```
